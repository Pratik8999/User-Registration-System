from user.models import CustomUser
from user.serializers import CustomUserSerializer
from rest_framework.parsers import JSONParser
from rest_framework.renderers import JSONRenderer
import io

pratik = CustomUser.objects.get(pk=1)


# Serializing object
serialized_object = CustomUserSerializer(pratik)
json_byte = JSONRenderer().render(serialized_object.data)

# Deserializing JSON Bytes

 


content = io.BytesIO(json_byte)
deserialized_data = JSONParser().parse(content)
convert_to_CustomUser = CustomUserSerializer(data=deserialized_data)
convert_to_CustomUser.is_valid()
